## 1로 만들기

```c++
#include<iostream>
using namespace std;
int D[100001] = {0, 0, 1, 1}; //index 1부터 시작

int main(){
    int N;
    cin >> N;

    for(int i = 4; i<=N; i++){
        // 1 빼는 것이 기본 연산.
        D[i] = D[i-1] +1;
        // 만약 3으로 나눠질 때, 1뺀 연산보다 D[i]값이 작으면 작은거로 바꿔준다.
        if(i%3 == 0){
            D[i] = min(D[i/3]+1, D[i]);
        }
        if(i%2 ==0){
        // 2로 나눠질 때와 2,3동시에 나눠질 때도 있으므로 if
            D[i] = min(D[i/2]+1, D[i]);
        }
    }

    cout << D[N];
}

/*
D[N] : N을 1로 만드는 연산의 최소 횟수
D[N] = min(D[N-1]+1,D[N/3]+1,D[N/2]+1)
*/
```
